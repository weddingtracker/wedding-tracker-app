<script>
document.addEventListener('DOMContentLoaded', () => {
const guestNameInput =
document.getElementById('guest-name-input');
const addGuestBtn =
document.getElementById('add-guest-btn');
const guestListUl = document.getElementById('guest-list');
const giftListUl = document.getElementById('gift-list');
const showGuestsBtn =
document.getElementById('show-guests-btn');
const showGiftsBtn =
document.getElementById('show-gifts-btn');
const guestsSection =
document.getElementById('guests-section');
const giftsSection =
document.getElementById('gifts-section');
const upgradePremiumBtn =
document.getElementById('upgrade-premium-btn');
const adBanner = document.getElementById('ad-banner');
const premiumAdContainer =
document.getElementById('premium-ad-container');
const interstitialModal =
document.getElementById('interstitial-modal');
const closeInterstitialBtn =
document.getElementById('close-interstitial-btn');
let isPremium = localStorage.getItem('isPremium') ===
'true';
const showInterstitial = () => {
// Only show ad if not a premium user
if (!isPremium) {
interstitialModal.classList.remove('hidden');
}
};
const togglePremiumUI = () => {
if (isPremium) {
adBanner.classList.add('hidden');
premiumAdContainer.classList.add('hidden');
} else {
adBanner.classList.remove('hidden');
premiumAdContainer.classList.remove('hidden');
}
};
const renderGuests = () => {
const guests =
JSON.parse(localStorage.getItem('guests')) || [];
guestListUl.innerHTML = '';
guests.forEach((guest, index) => {
const li = document.createElement('li');
li.className = "flex items-center justify-between
p-3 bg-gray-50 rounded-lg shadow-sm";
li.innerHTML = `
<div class="flex-1">
<span class="text-gray-800
font-medium">${guest.name}</span>
<span class="text-sm text-gray-500
ml-2">(${guest.status})</span>
</div>
<div class="flex space-x-2">
<button class="status-btn text-xs
font-semibold py-1 px-3 rounded-full ${guest.status === 'RSVP' ?
'bg-green-200 text-green-700' : 'bg-gray-200 text-gray-700'}"
data-index="${index}" data-status="RSVP">RSVP</button>
<button class="status-btn text-xs
font-semibold py-1 px-3 rounded-full ${guest.status === 'Declined' ?
'bg-red-200 text-red-700' : 'bg-gray-200 text-gray-700'}"
data-index="${index}" data-status="Declined">Declined</button>
<button class="delete-btn text-red-500
hover:text-red-700 transition duration-300" data-index="${index}">
<svg
xmlns="http://www.w3.org/2000/svg" class="h-5 w-5" viewBox="0 0 20 20"
fill="currentColor">
<path fill-rule="evenodd" d="M9
2a1 1 0 00-.894.553L7.382 4H4a1 1 0 000 2v10a2 2 0 002 2h8a2 2 0
002-2V6a1 1 0 100-2h-3.382l-.724-1.447A1 1 0 0011 2H9zM7 8a1 1 0 012
0v6a1 1 0 11-2 0V8zm6 0a1 1 0 012 0v6a1 1 0 11-2 0V8z"
clip-rule="evenodd" />
</svg>
</button>
</div>
`;
guestListUl.appendChild(li);
});
};
const renderGifts = () => {
const gifts =
JSON.parse(localStorage.getItem('gifts')) || [];
giftListUl.innerHTML = '';
if (gifts.length === 0) {
giftListUl.innerHTML = `<li class="text-center
text-gray-500 py-4">No gifts tracked yet.</li>`;
}
gifts.forEach((gift, index) => {
const li = document.createElement('li');
li.className = "flex items-center justify-between
p-3 bg-gray-50 rounded-lg shadow-sm";
li.innerHTML = `
<div class="flex-1">
<span class="text-gray-800
font-medium">${gift.name}</span>
<span class="text-sm text-gray-500
ml-2">(${gift.from})</span>
</div>
<button class="delete-gift-btn text-red-500
hover:text-red-700 transition duration-300" data-index="${index}">
<svg xmlns="http://www.w3.org/2000/svg"
class="h-5 w-5" viewBox="0 0 20 20" fill="currentColor">
<path fill-rule="evenodd" d="M9 2a1 1
0 00-.894.553L7.382 4H4a1 1 0 000 2v10a2 2 0 002 2h8a2 2 0 002-2V6a1 1
0 100-2h-3.382l-.724-1.447A1 1 0 0011 2H9zM7 8a1 1 0 012 0v6a1 1 0
11-2 0V8zm6 0a1 1 0 012 0v6a1 1 0 11-2 0V8z" clip-rule="evenodd" />
</svg>
</button>
`;
giftListUl.appendChild(li);
});
};
const addGuest = () => {
const guestName = guestNameInput.value.trim();
if (guestName) {
const guests =
JSON.parse(localStorage.getItem('guests')) || [];
guests.push({ name: guestName, status: 'Not
RSVPed' });
localStorage.setItem('guests',
JSON.stringify(guests));
guestNameInput.value = '';
renderGuests();
}
};
const handleGuestAction = (event) => {
const target = event.target;
const index = target.getAttribute('data-index');
const guests =
JSON.parse(localStorage.getItem('guests'));
if (target.classList.contains('status-btn')) {
const status = target.getAttribute('data-status');
guests[index].status = status;
localStorage.setItem('guests',
JSON.stringify(guests));
showInterstitial(); // Show ad after a status
change
} else if (target.closest('.delete-btn')) {
guests.splice(index, 1);
localStorage.setItem('guests',
JSON.stringify(guests));
}
renderGuests();
};
const handleGiftAction = (event) => {
const target = event.target;
if (target.closest('.delete-gift-btn')) {
const index =
target.closest('.delete-gift-btn').getAttribute('data-index');
const gifts =
JSON.parse(localStorage.getItem('gifts'));
gifts.splice(index, 1);
localStorage.setItem('gifts',
JSON.stringify(gifts));
renderGifts();
}
};
const handleUpgrade = () => {
localStorage.setItem('isPremium', 'true');
isPremium = true;
togglePremiumUI();
// Show a confirmation message instead of an alert
const modal = document.createElement('div');
modal.className = "modal";
modal.innerHTML = `
<div class="modal-content">
<h3 class="text-xl font-bold mb-4
text-green-600">Upgrade Successful!</h3>
<p class="text-gray-700 mb-6">Thank you for
your purchase. All ads have been removed.</p>
<button id="close-modal-btn"
class="bg-indigo-500 text-white font-bold py-3 px-8 rounded-full
hover:bg-indigo-600 transition duration-300">Close</button>
</div>
`;
document.body.appendChild(modal);
document.getElementById('close-modal-btn').addEventListener('click',
() => {
modal.remove();
});
};
// Set up initial UI state
togglePremiumUI();
renderGuests();
renderGifts();
// Event Listeners
addGuestBtn.addEventListener('click', addGuest);
guestNameInput.addEventListener('keypress', (e) => {
if (e.key === 'Enter') {
addGuest();
}
});
guestListUl.addEventListener('click', handleGuestAction);
giftListUl.addEventListener('click', handleGiftAction);
showGuestsBtn.addEventListener('click', () => {
guestsSection.classList.remove('hidden');
giftsSection.classList.add('hidden');
showGuestsBtn.classList.remove('bg-gray-200',
'text-gray-700');
showGuestsBtn.classList.add('bg-indigo-500',
'text-white');
showGiftsBtn.classList.remove('bg-indigo-500',
'text-white');
showGiftsBtn.classList.add('bg-gray-200',
'text-gray-700');
});
showGiftsBtn.addEventListener('click', () => {
guestsSection.classList.add('hidden');
giftsSection.classList.remove('hidden');
showGiftsBtn.classList.remove('bg-gray-200',
'text-gray-700');
showGiftsBtn.classList.add('bg-indigo-500',
'text-white');
showGuestsBtn.classList.remove('bg-indigo-500',
'text-white');
showGuestsBtn.classList.add('bg-gray-200',
'text-gray-700');
});
upgradePremiumBtn.addEventListener('click',
handleUpgrade);
closeInterstitialBtn.addEventListener('click', () => {
interstitialModal.classList.add('hidden');
});
// Simulate a few guests and gifts for demonstration
if (!localStorage.getItem('guests')) {
const initialGuests = [
{ name: 'John Doe', status: 'Not RSVPed' },
{ name: 'Jane Smith', status: 'RSVP' },
{ name: 'Peter Jones', status: 'Declined' }
];
localStorage.setItem('guests',
JSON.stringify(initialGuests));
renderGuests();
}
if (!localStorage.getItem('gifts')) {
const initialGifts = [
{ name: 'Blender', from: 'Aunt Carol' },
{ name: 'Cash', from: 'Grandma Helen' }
];
localStorage.setItem('gifts',
JSON.stringify(initialGifts));
renderGifts();
}
});
</script>
</body>
